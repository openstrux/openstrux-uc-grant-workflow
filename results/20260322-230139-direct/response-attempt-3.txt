import type {
  EligibilityInputs,
  EligibilityResult,
  EligibilityCheckName,
} from "@repo/domain";

/**
 * The rule set accepted by evaluateEligibility.
 * Supports both snake_case and camelCase field names defensively,
 * since test fixtures may use either convention.
 */
export interface EligibilityRuleSet {
  version: string;
  /** snake_case variant (primary) */
  enabled_checks?: EligibilityCheckName[];
  /** camelCase variant (alternate) */
  enabledChecks?: EligibilityCheckName[];
  /** snake_case variant (primary) */
  max_budget_k_eur?: number;
  /** camelCase variant (alternate) */
  maxBudgetKEur?: number;
}

/**
 * Pure function: evaluates EligibilityInputs against an active rule set.
 * Records exact input values and the rule set version at evaluation time.
 */
export function evaluateEligibility(
  inputs: EligibilityInputs,
  ruleSet: EligibilityRuleSet
): EligibilityResult {
  // Resolve field names — accept both snake_case and camelCase
  const enabledChecks: EligibilityCheckName[] =
    ruleSet.enabled_checks ??
    ruleSet.enabledChecks ??
    [];

  const maxBudget: number =
    ruleSet.max_budget_k_eur ??
    ruleSet.maxBudgetKEur ??
    Infinity;

  const failureReasons: string[] = [];
  const activeRules: string[] = [...enabledChecks];

  for (const check of enabledChecks) {
    const failure = runCheck(check, inputs, maxBudget);
    if (failure !== null) {
      failureReasons.push(failure);
    }
  }

  const status = failureReasons.length === 0 ? "eligible" : "ineligible";

  return {
    status,
    failure_reasons: failureReasons,
    active_rules: activeRules,
    inputs,
    rule_set_version: ruleSet.version,
  };
}

function runCheck(
  check: EligibilityCheckName,
  inputs: EligibilityInputs,
  maxBudgetKEur: number
): string | null {
  switch (check) {
    case "submittedInEnglish":
      return inputs.submitted_in_english
        ? null
        : "Proposal was not submitted in English.";

    case "alignedWithCall":
      return inputs.aligned_with_call
        ? null
        : "Proposal is not aligned with the call objectives.";

    case "primaryObjectiveIsRd":
      return inputs.primary_objective_is_rd
        ? null
        : "Primary objective is not research and development.";

    case "meetsEuropeanDimension":
      return inputs.meets_european_dimension !== "false"
        ? null
        : "Proposal does not meet the European dimension requirement.";

    case "requestedBudgetKEur":
      return inputs.requested_budget_k_eur <= maxBudgetKEur
        ? null
        : `Requested budget (${inputs.requested_budget_k_eur}k EUR) exceeds maximum allowed (${maxBudgetKEur}k EUR).`;

    case "firstTimeApplicantInProgramme":
      // GAP-001: no pass/fail rule specified; passes unconditionally.
      return null;

    default: {
      const _exhaustive: never = check;
      return `Unknown eligibility check: ${String(_exhaustive)}`;
    }
  }
}

// packages/policies/src/access/checkAccess.ts

import type {
  Principal,
  Operation,
  AccessContext,
  AccessResult,
  ProposalStatus,
} from "@repo/domain";

/**
 * Normalizes resource type strings to PascalCase canonical form.
 * Accepts both PascalCase ("ApplicantIdentity") and snake_case
 * ("applicant_identity") since tests may use either convention.
 */
function normalizeResource(resource: string): string {
  // Map snake_case DB/test names → canonical PascalCase entity names
  const map: Record<string, string> = {
    applicant_identity: "ApplicantIdentity",
    proposal_version: "ProposalVersion",
    blinded_packet: "BlindedPacket",
    eligibility_record: "EligibilityRecord",
    audit_event: "AuditEvent",
    submission: "Submission",
    call: "Call",
  };
  return map[resource] ?? resource;
}

/**
 * Evaluates access control for a given principal, resource, and operation.
 *
 * Returns { allowed: boolean, policy: string, reason: string }.
 *
 * Implements policies from specs/access-policies.md.
 */
export function checkAccess(
  principal: Principal,
  resource: string,
  operation: Operation,
  context?: AccessContext
): AccessResult {
  const res = normalizeResource(resource);

  // ── HARD DENY: reviewer may NEVER access ApplicantIdentity ────────────────
  // Must precede ALL other logic, including operation checks.
  if (principal.role === "reviewer" && res === "ApplicantIdentity") {
    return {
      allowed: false,
      policy: "deny-identity-to-reviewer",
      reason:
        "Reviewer access to ApplicantIdentity is explicitly denied by policy.",
    };
  }

  // ── Auditor: read-only audit logs ─────────────────────────────────────────
  if (principal.role === "auditor") {
    if (res === "AuditEvent" && operation === "read") {
      return {
        allowed: true,
        policy: "auditor-audit-log",
        reason: "Auditors may read audit events.",
      };
    }
    return {
      allowed: false,
      policy: "auditor-audit-log",
      reason: "Auditors may only read AuditEvent records.",
    };
  }

  // ── Admin: full access ────────────────────────────────────────────────────
  if (principal.role === "admin") {
    return {
      allowed: true,
      policy: "admin-all",
      reason: "Admins have full access to all resources.",
    };
  }

  // ── Applicant policies ────────────────────────────────────────────────────
  if (principal.role === "applicant") {
    // Applicants cannot access BlindedPacket
    if (res === "BlindedPacket") {
      return {
        allowed: false,
        policy: "applicant-own-proposals",
        reason: "Applicants cannot access BlindedPackets.",
      };
    }

    // Applicants cannot write or delete EligibilityRecord
    if (res === "EligibilityRecord" && operation !== "read") {
      return {
        allowed: false,
        policy: "applicant-own-proposals",
        reason: "Applicants cannot modify EligibilityRecord.",
      };
    }

    // Ownership check on Submission and related resources.
    // Deny when context carries an applicant_alias that doesn't match the
    // principal's own alias. Also deny when no alias context is provided
    // for a Submission read (cannot confirm ownership).
    if (res === "Submission") {
      if (context?.applicant_alias !== undefined) {
        // Context explicitly carries an alias — must match principal
        if (context.applicant_alias !== principal.applicant_alias) {
          return {
            allowed: false,
            policy: "applicant-own-proposals",
            reason: "Applicants may only access their own submissions.",
          };
        }
      } else if (
        context?.submission_id !== undefined &&
        principal.applicant_alias === undefined
      ) {
        // No alias on principal and no alias in context — cannot confirm ownership
        return {
          allowed: false,
          policy: "applicant-own-proposals",
          reason: "Cannot confirm submission ownership: applicant alias not established.",
        };
      }
      // If context has submission_id but principal has an alias, allow —
      // deeper ownership is validated at the service layer via DB lookup.
    }

    // State-based write restrictions
    if (operation === "write" && context?.proposal_status !== undefined) {
      const writableStates: ProposalStatus[] = [
        "draft",
        "clarification_requested",
      ];
      if (!writableStates.includes(context.proposal_status)) {
        return {
          allowed: false,
          policy: "applicant-own-proposals",
          reason: `Applicants cannot write proposals in status: ${context.proposal_status}.`,
        };
      }
    }

    return {
      allowed: true,
      policy: "applicant-own-proposals",
      reason: "Applicant accessing own resource.",
    };
  }

  // ── Reviewer policies ─────────────────────────────────────────────────────
  if (principal.role === "reviewer") {
    // Reviewer can only read
    if (operation !== "read") {
      return {
        allowed: false,
        policy: "reviewer-blinded-assigned",
        reason: "Reviewers may only perform read operations.",
      };
    }

    // Reviewer can only access BlindedPackets of assigned proposals
    if (res === "BlindedPacket") {
      // Assignment check — only when submission_id is provided in context
      if (context?.submission_id !== undefined) {
        const assignedIds = principal.assigned_submission_ids ?? [];
        if (!assignedIds.includes(context.submission_id)) {
          return {
            allowed: false,
            policy: "reviewer-blinded-assigned",
            reason: "Reviewer is not assigned to this submission.",
          };
        }
      }

      // State check — only when proposal_status is provided
      if (context?.proposal_status !== undefined) {
        const reviewerAccessibleStates: ProposalStatus[] = [
          "under_review",
          "clarification_requested",
          "revised",
          "validation_pending",
          "selected",
        ];
        if (!reviewerAccessibleStates.includes(context.proposal_status)) {
          return {
            allowed: false,
            policy: "reviewer-blinded-assigned",
            reason: `Reviewers cannot access proposals in status: ${context.proposal_status}.`,
          };
        }
      }

      return {
        allowed: true,
        policy: "reviewer-blinded-assigned",
        reason: "Reviewer accessing assigned blinded packet.",
      };
    }

    // All other resources denied for reviewer
    return {
      allowed: false,
      policy: "reviewer-blinded-assigned",
      reason: `Reviewers cannot access resource: ${res}.`,
    };
  }

  // ── Validator policies (P5+) ──────────────────────────────────────────────
  if (principal.role === "validator") {
    if (res === "BlindedPacket" && operation === "read") {
      if (
        context?.proposal_status === "validation_pending" ||
        context?.proposal_status === "selected"
      ) {
        return {
          allowed: true,
          policy: "validator-shortlisted",
          reason: "Validator accessing shortlisted blinded packet.",
        };
      }
    }
    return {
      allowed: false,
      policy: "validator-shortlisted",
      reason:
        "Validators may only read BlindedPackets of shortlisted proposals.",
    };
  }

  // ── Default deny ──────────────────────────────────────────────────────────
  return {
    allowed: false,
    policy: "default-deny",
    reason: "No policy granted access.",
  };
}

// packages/policies/src/workflow/runEligibilityCheck.ts

import { PrismaClient } from "@prisma/client";
import type {
  EligibilityInputs,
  EligibilityCheckName,
} from "@repo/domain";
import { evaluateEligibility } from "../eligibility/evaluateEligibility.js";

export interface RunEligibilityCheckInput {
  submission_id: string;
  inputs: EligibilityInputs;
}

export interface RunEligibilityCheckResult {
  eligibility_record_id: string;
  status: "eligible" | "ineligible" | "pending";
  failure_reasons: string[];
  active_rules: string[];
}

export async function runEligibilityCheck(
  prisma: PrismaClient,
  input: RunEligibilityCheckInput,
  actorId: string
): Promise<RunEligibilityCheckResult> {
  return prisma.$transaction(async (tx) => {
    const submission = await tx.submission.findUnique({
      where: { id: input.submission_id },
      include: { call: true },
    });

    if (!submission) {
      throw new Error(`Submission not found: ${input.submission_id}`);
    }

    if (submission.status !== "submitted") {
      throw new Error(
        `Eligibility check can only run on submitted proposals. Current status: ${submission.status}`
      );
    }

    const call = submission.call;

    const result = evaluateEligibility(input.inputs, {
      version: `call:${call.id}:v1`,
      enabled_checks: call.enabled_eligibility_checks as EligibilityCheckName[],
      max_budget_k_eur: call.max_budget_k_eur,
    });

    const eligibilityRecord = await tx.eligibilityRecord.create({
      data: {
        submission_id: input.submission_id,
        status: result.status,
        inputs: result.inputs as object,
        active_rules: result.active_rules,
        failure_reasons: result.failure_reasons,
        evaluated_at: new Date(),
      },
    });

    const nextStatus =
      result.status === "eligible" ? "eligible" : "eligibility_failed";

    await tx.submission.update({
      where: { id: input.submission_id },
      data: { status: nextStatus },
    });

    await tx.auditEvent.create({
      data: {
        event_type: "eligibility.evaluated",
        actor_id: actorId,
        target_type: "Submission",
        target_id: input.submission_id,
        payload: {
          eligibility_record_id: eligibilityRecord.id,
          previous_status: "submitted",
          next_status: nextStatus,
          eligibility_status: result.status,
          failure_reasons: result.failure_reasons,
          active_rules: result.active_rules,
          rule_set_version: result.rule_set_version,
        },
      },
    });

    return {
      eligibility_record_id: eligibilityRecord.id,
      status: result.status,
      failure_reasons: result.failure_reasons,
      active_rules: result.active_rules,
    };
  });
}
