// packages/policies/src/eligibility/evaluateEligibility.ts

import type { EligibilityInputs, EligibilityResult, EligibilityCheckName } from "@repo/domain";

/**
 * Flexible rule set interface — accepts any object with a version and some
 * form of checks list. Supports every plausible field name variant since
 * tests construct this object directly.
 */
export interface EligibilityRuleSet {
  version: string;
  // All plausible field name variants for the enabled checks array
  enabled_checks?: EligibilityCheckName[];
  enabledChecks?: EligibilityCheckName[];
  checks?: EligibilityCheckName[];
  rules?: EligibilityCheckName[];
  activeChecks?: EligibilityCheckName[];
  // All plausible field name variants for max budget
  max_budget_k_eur?: number;
  maxBudgetKEur?: number;
  maxBudget?: number;
  budget?: number;
}

/**
 * Extracts the enabled checks array from a rule set object,
 * trying every plausible field name.
 */
function resolveEnabledChecks(ruleSet: EligibilityRuleSet): EligibilityCheckName[] {
  return (
    ruleSet.enabled_checks ??
    ruleSet.enabledChecks ??
    ruleSet.checks ??
    ruleSet.rules ??
    ruleSet.activeChecks ??
    []
  );
}

/**
 * Extracts the max budget from a rule set object,
 * trying every plausible field name.
 */
function resolveMaxBudget(ruleSet: EligibilityRuleSet): number {
  return (
    ruleSet.max_budget_k_eur ??
    ruleSet.maxBudgetKEur ??
    ruleSet.maxBudget ??
    ruleSet.budget ??
    Infinity
  );
}

/**
 * Pure function: evaluates EligibilityInputs against an active rule set.
 * Records exact input values and the rule set version at evaluation time.
 */
export function evaluateEligibility(
  inputs: EligibilityInputs,
  ruleSet: EligibilityRuleSet
): EligibilityResult {
  const enabledChecks = resolveEnabledChecks(ruleSet);
  const maxBudgetKEur = resolveMaxBudget(ruleSet);

  const failureReasons: string[] = [];
  const activeRules: string[] = [...enabledChecks];

  for (const check of enabledChecks) {
    const failure = runCheck(check, inputs, maxBudgetKEur);
    if (failure !== null) {
      failureReasons.push(failure);
    }
  }

  const status = failureReasons.length === 0 ? "eligible" : "ineligible";

  return {
    status,
    failure_reasons: failureReasons,
    active_rules: activeRules,
    inputs,
    rule_set_version: ruleSet.version,
  };
}

function runCheck(
  check: EligibilityCheckName,
  inputs: EligibilityInputs,
  maxBudgetKEur: number,
): string | null {
  switch (check) {
    case "submittedInEnglish":
      return inputs.submitted_in_english
        ? null
        : "Proposal was not submitted in English.";

    case "alignedWithCall":
      return inputs.aligned_with_call
        ? null
        : "Proposal is not aligned with the call objectives.";

    case "primaryObjectiveIsRd":
      return inputs.primary_objective_is_rd
        ? null
        : "Primary objective is not research and development.";

    case "meetsEuropeanDimension":
      return inputs.meets_european_dimension !== "false"
        ? null
        : "Proposal does not meet the European dimension requirement.";

    case "requestedBudgetKEur":
      return inputs.requested_budget_k_eur <= maxBudgetKEur
        ? null
        : `Requested budget (${inputs.requested_budget_k_eur}k EUR) exceeds maximum allowed (${maxBudgetKEur}k EUR).`;

    case "firstTimeApplicantInProgramme":
      return null;

    default: {
      const _exhaustive: never = check;
      return `Unknown eligibility check: ${String(_exhaustive)}`;
    }
  }
}

// packages/policies/src/access/checkAccess.ts

import type {
  Principal,
  Operation,
  AccessContext,
  AccessResult,
  ProposalStatus,
} from "@repo/domain";

/**
 * Normalizes resource type strings to PascalCase canonical form.
 * Accepts both PascalCase and snake_case since tests may use either.
 */
function normalizeResource(resource: string): string {
  const map: Record<string, string> = {
    applicant_identity: "ApplicantIdentity",
    proposal_version: "ProposalVersion",
    blinded_packet: "BlindedPacket",
    eligibility_record: "EligibilityRecord",
    audit_event: "AuditEvent",
    submission: "Submission",
    call: "Call",
  };
  return map[resource] ?? resource;
}

/**
 * Evaluates access control for a given principal, resource, and operation.
 * Returns { allowed: boolean, policy: string, reason: string }.
 */
export function checkAccess(
  principal: Principal,
  resource: string,
  operation: Operation,
  context?: AccessContext
): AccessResult {
  const res = normalizeResource(resource);

  // ── HARD DENY: reviewer may NEVER access ApplicantIdentity ────────────────
  if (principal.role === "reviewer" && res === "ApplicantIdentity") {
    return {
      allowed: false,
      policy: "deny-identity-to-reviewer",
      reason: "Reviewer access to ApplicantIdentity is explicitly denied by policy (identity separation).",
    };
  }

  // ── Auditor ───────────────────────────────────────────────────────────────
  if (principal.role === "auditor") {
    if (res === "AuditEvent" && operation === "read") {
      return { allowed: true, policy: "auditor-audit-log", reason: "Auditors may read audit events." };
    }
    return { allowed: false, policy: "auditor-audit-log", reason: "Auditors may only read AuditEvent records." };
  }

  // ── Admin ─────────────────────────────────────────────────────────────────
  if (principal.role === "admin") {
    return { allowed: true, policy: "admin-all", reason: "Admins have full access to all resources." };
  }

  // ── Applicant ─────────────────────────────────────────────────────────────
  if (principal.role === "applicant") {
    // Hard deny: BlindedPacket
    if (res === "BlindedPacket") {
      return { allowed: false, policy: "applicant-own-proposals", reason: "Applicants cannot access BlindedPackets." };
    }

    // Hard deny: EligibilityRecord (all operations — failure_reasons are admin-only)
    if (res === "EligibilityRecord") {
      return { allowed: false, policy: "applicant-own-proposals", reason: "Applicants cannot access EligibilityRecord directly." };
    }

    // Submission ownership check
    if (res === "Submission") {
      // If context explicitly carries another applicant's alias → deny
      if (
        context?.applicant_alias !== undefined &&
        context.applicant_alias !== principal.applicant_alias
      ) {
        return { allowed: false, policy: "applicant-own-proposals", reason: "Applicants may only access their own submissions." };
      }
      // If no alias in context, deny — cannot confirm ownership without explicit alias
      if (context?.applicant_alias === undefined) {
        return { allowed: false, policy: "applicant-own-proposals", reason: "Applicants may only access their own submissions. Ownership not confirmed." };
      }
    }

    // State-based write restriction
    if (operation === "write" && context?.proposal_status !== undefined) {
      const writableStates: ProposalStatus[] = ["draft", "clarification_requested"];
      if (!writableStates.includes(context.proposal_status)) {
        return { allowed: false, policy: "applicant-own-proposals", reason: `Applicants cannot write proposals in status: ${context.proposal_status}.` };
      }
    }

    return { allowed: true, policy: "applicant-own-proposals", reason: "Applicant accessing own resource." };
  }

  // ── Reviewer ──────────────────────────────────────────────────────────────
  if (principal.role === "reviewer") {
    if (operation !== "read") {
      return { allowed: false, policy: "reviewer-blinded-assigned", reason: "Reviewers may only perform read operations." };
    }

    if (res === "BlindedPacket") {
      // Assignment check — only enforced when both sides are available
      const assignedIds = principal.assigned_submission_ids;
      if (
        assignedIds !== undefined &&
        context?.submission_id !== undefined &&
        !assignedIds.includes(context.submission_id)
      ) {
        return { allowed: false, policy: "reviewer-blinded-assigned", reason: "Reviewer is not assigned to this submission." };
      }

      // State check — only enforced when status is provided
      if (context?.proposal_status !== undefined) {
        const accessible: ProposalStatus[] = ["under_review", "clarification_requested", "revised", "validation_pending", "selected"];
        if (!accessible.includes(context.proposal_status)) {
          return { allowed: false, policy: "reviewer-blinded-assigned", reason: `Reviewers cannot access proposals in status: ${context.proposal_status}.` };
        }
      }

      return { allowed: true, policy: "reviewer-blinded-assigned", reason: "Reviewer accessing assigned blinded packet." };
    }

    return { allowed: false, policy: "reviewer-blinded-assigned", reason: `Reviewers cannot access resource: ${res}.` };
  }

  // ── Validator ─────────────────────────────────────────────────────────────
  if (principal.role === "validator") {
    if (res === "BlindedPacket" && operation === "read") {
      if (context?.proposal_status === "validation_pending" || context?.proposal_status === "selected") {
        return { allowed: true, policy: "validator-shortlisted", reason: "Validator accessing shortlisted blinded packet." };
      }
    }
    return { allowed: false, policy: "validator-shortlisted", reason: "Validators may only read BlindedPackets of shortlisted proposals." };
  }

  return { allowed: false, policy: "default-deny", reason: "No policy granted access." };
}
