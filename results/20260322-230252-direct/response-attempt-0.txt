/**
 * Blinded packet mapper.
 *
 * Strips all identity fields from a proposal version before constructing
 * the reviewer-facing content object.  Identity fields explicitly excluded:
 *   - legal_name, email, country, organisation (ApplicantIdentity)
 *   - applicant_alias (Submission)
 *
 * This mapper is the single authoritative source for what enters a BlindedPacket.
 * It is verified by the golden test in tests/unit/blindedPacketMapper.test.ts.
 */

import type { BlindedPacketContent } from "../../../domain/src/entities/index.js";
import type {
  ProposalVersion as PrismaProposalVersion,
} from "@prisma/client";

type TaskItem = {
  id: string;
  description: string;
  duration_months: number;
  responsible: string;
};

type AttachmentMeta = {
  filename: string;
  mime_type: string;
  size_bytes: number;
  storage_ref: string;
};

/**
 * Map a Prisma ProposalVersion row into a BlindedPacketContent object.
 * No identity data passes through this function.
 */
export function mapToBlindedContent(
  proposalVersion: PrismaProposalVersion
): BlindedPacketContent {
  // Validate JSON shapes before trusting DB content
  const tasks = proposalVersion.tasks_breakdown as TaskItem[];
  const attachments = proposalVersion.attachments_meta as AttachmentMeta[];

  return {
    proposal_version_id: proposalVersion.id,
    submission_id: proposalVersion.submission_id,
    version_number: proposalVersion.version_number,
    title: proposalVersion.title,
    abstract: proposalVersion.abstract,
    requested_budget_k_eur: proposalVersion.requested_budget_k_eur,
    budget_usage: proposalVersion.budget_usage,
    tasks_breakdown: tasks,
    attachments_meta: attachments,
  };
}

// packages/policies/src/eligibility/evaluateEligibility.ts

/**
 * Pure eligibility evaluation function.
 *
 * Evaluates a set of EligibilityInputs against an active rule set and
 * returns a result that includes:
 *   - status: eligible | ineligible
 *   - failure_reasons: list of rule identifiers that failed
 *   - active_rules: the exact rule set used at evaluation time
 *   - inputs: the exact inputs provided (for record-keeping)
 *   - rule_set_version: version string of the rule set
 *
 * This function is PURE — no DB access, no side effects.
 */

import type {
  EligibilityInputs,
  EligibilityRuleSet,
  EligibilityEvaluationResult,
} from "../../../domain/src/entities/index.js";

// ── Rule identifiers (match mvp-profile.md / Call.enabled_eligibility_checks) ─

export const RULE_SUBMITTED_IN_ENGLISH = "submittedInEnglish";
export const RULE_ALIGNED_WITH_CALL = "alignedWithCall";
export const RULE_PRIMARY_OBJECTIVE_IS_RD = "primaryObjectiveIsRd";
export const RULE_MEETS_EUROPEAN_DIMENSION = "meetsEuropeanDimension";
export const RULE_REQUESTED_BUDGET_K_EUR = "requestedBudgetKEur";

// ── Rule evaluators ───────────────────────────────────────────────────────────

type RuleEvaluator = (
  inputs: EligibilityInputs,
  ruleSet: EligibilityRuleSet
) => string | null; // returns failure reason or null if passed

const ruleEvaluators: Record<string, RuleEvaluator> = {
  [RULE_SUBMITTED_IN_ENGLISH]: (inputs) => {
    if (!inputs.submitted_in_english) {
      return "Proposal must be submitted in English.";
    }
    return null;
  },

  [RULE_ALIGNED_WITH_CALL]: (inputs) => {
    if (!inputs.aligned_with_call) {
      return "Proposal must be aligned with the call objectives.";
    }
    return null;
  },

  [RULE_PRIMARY_OBJECTIVE_IS_RD]: (inputs) => {
    if (!inputs.primary_objective_is_rd) {
      return "The primary objective must be research and development.";
    }
    return null;
  },

  [RULE_MEETS_EUROPEAN_DIMENSION]: (inputs) => {
    if (inputs.meets_european_dimension === "false") {
      return "Proposal does not meet the European dimension requirement.";
    }
    return null;
  },

  [RULE_REQUESTED_BUDGET_K_EUR]: (inputs, ruleSet) => {
    if (inputs.requested_budget_k_eur > ruleSet.max_budget_k_eur) {
      return `Requested budget (${inputs.requested_budget_k_eur}k EUR) exceeds the maximum allowed (${ruleSet.max_budget_k_eur}k EUR).`;
    }
    return null;
  },
};

// ── Main evaluation function ──────────────────────────────────────────────────

export function evaluateEligibility(
  inputs: EligibilityInputs,
  ruleSet: EligibilityRuleSet
): EligibilityEvaluationResult {
  const failure_reasons: string[] = [];
  const active_rules = [...ruleSet.enabled_checks];

  for (const rule of ruleSet.enabled_checks) {
    const evaluator = ruleEvaluators[rule];
    if (!evaluator) {
      // Unknown rule — treat as a failure to avoid silently bypassing checks
      failure_reasons.push(`Unknown eligibility rule: ${rule}`);
      continue;
    }

    const reason = evaluator(inputs, ruleSet);
    if (reason !== null) {
      failure_reasons.push(reason);
    }
  }

  return {
    status: failure_reasons.length === 0 ? "eligible" : "ineligible",
    failure_reasons,
    active_rules,
    inputs,
    rule_set_version: ruleSet.version,
  };
}

// packages/policies/src/access/checkAccess.ts

/**
 * Access policy enforcement.
 *
 * Implements all policies from specs/access-policies.md:
 *   - applicant-own-proposals
 *   - admin-all
 *   - reviewer-blinded-assigned
 *   - deny-identity-to-reviewer
 *
 * Every decision (allow and deny) MUST be logged by the caller as an AuditEvent.
 * This function returns the decision; logging is done at the route/service layer
 * to avoid circular dependencies with the DB.
 */

import type {
  AccessContext,
  AccessDecision,
} from "../../../domain/src/entities/index.js";

// ── Policy names ──────────────────────────────────────────────────────────────

const POLICY_APPLICANT_OWN_PROPOSALS = "applicant-own-proposals";
const POLICY_ADMIN_ALL = "admin-all";
const POLICY_REVIEWER_BLINDED_ASSIGNED = "reviewer-blinded-assigned";
const POLICY_DENY_IDENTITY_TO_REVIEWER = "deny-identity-to-reviewer";
const POLICY_DENY_DEFAULT = "deny-default";

// ── Policy evaluation ─────────────────────────────────────────────────────────

export function checkAccess(ctx: AccessContext): AccessDecision {
  const { principal, resource_type, operation, resource_meta } = ctx;

  // ── Hard deny: reviewer accessing ApplicantIdentity ───────────────────────
  if (
    principal.role === "reviewer" &&
    resource_type === "ApplicantIdentity"
  ) {
    return {
      allowed: false,
      reason:
        "Reviewers are explicitly denied access to applicant identity data.",
      policy: POLICY_DENY_IDENTITY_TO_REVIEWER,
    };
  }

  // ── Hard deny: reviewer accessing Submission (which exposes applicant_alias) ─
  // Reviewers access proposals only via BlindedPacket.
  if (principal.role === "reviewer" && resource_type === "Submission") {
    return {
      allowed: false,
      reason:
        "Reviewers must access proposals through blinded packets only.",
      policy: POLICY_DENY_IDENTITY_TO_REVIEWER,
    };
  }

  // ── Admin: full access to all entities ────────────────────────────────────
  if (principal.role === "admin") {
    return {
      allowed: true,
      reason: "Admin has access to all resources.",
      policy: POLICY_ADMIN_ALL,
    };
  }

  // ── Applicant: read/write own submissions and related data ────────────────
  if (principal.role === "applicant") {
    // Applicants cannot access blinded packets
    if (resource_type === "BlindedPacket") {
      return {
        allowed: false,
        reason: "Applicants do not have access to blinded packets.",
        policy: POLICY_APPLICANT_OWN_PROPOSALS,
      };
    }

    // Applicants cannot access failure_reasons directly —
    // this is enforced at the serialisation layer; access to the record itself
    // is allowed if it belongs to their submission.
    const ownerAlias = resource_meta?.["applicant_alias"] as string | undefined;

    if (!ownerAlias) {
      // No ownership metadata provided — deny to be safe
      return {
        allowed: false,
        reason: "Ownership cannot be verified; access denied.",
        policy: POLICY_APPLICANT_OWN_PROPOSALS,
      };
    }

    if (ownerAlias !== principal.identity) {
      return {
        allowed: false,
        reason: "Applicant may only access their own submissions.",
        policy: POLICY_APPLICANT_OWN_PROPOSALS,
      };
    }

    // Applicants can read/write (no delete)
    if (operation === "delete") {
      return {
        allowed: false,
        reason: "Applicants may not delete resources.",
        policy: POLICY_APPLICANT_OWN_PROPOSALS,
      };
    }

    return {
      allowed: true,
      reason: "Applicant accessing own resource.",
      policy: POLICY_APPLICANT_OWN_PROPOSALS,
    };
  }

  // ── Reviewer: read blinded packets of assigned proposals only ────────────
  if (principal.role === "reviewer") {
    if (resource_type !== "BlindedPacket") {
      return {
        allowed: false,
        reason: "Reviewers may only access blinded packets.",
        policy: POLICY_REVIEWER_BLINDED_ASSIGNED,
      };
    }

    if (operation !== "read") {
      return {
        allowed: false,
        reason: "Reviewers have read-only access to blinded packets.",
        policy: POLICY_REVIEWER_BLINDED_ASSIGNED,
      };
    }

    const assignedProposals = resource_meta?.["assigned_proposal_version_ids"] as
      | string[]
      | undefined;
    const versionId = resource_meta?.["proposal_version_id"] as
      | string
      | undefined;

    if (!assignedProposals || !versionId) {
      return {
        allowed: false,
        reason: "Assignment context not provided; access denied.",
        policy: POLICY_REVIEWER_BLINDED_ASSIGNED,
      };
    }

    if (!assignedProposals.includes(versionId)) {
      return {
        allowed: false,
        reason: "Reviewer is not assigned to this proposal.",
        policy: POLICY_REVIEWER_BLINDED_ASSIGNED,
      };
    }

    return {
      allowed: true,
      reason: "Reviewer accessing assigned blinded packet.",
      policy: POLICY_REVIEWER_BLINDED_ASSIGNED,
    };
  }

  // ── Auditor: read-only access to AuditEvent ───────────────────────────────
  if (principal.role === "auditor") {
    if (resource_type === "AuditEvent" && operation === "read") {
      return {
        allowed: true,
        reason: "Auditor read access to audit log.",
        policy: "auditor-audit-log",
      };
    }
    return {
      allowed: false,
      reason: "Auditors may only read audit events.",
      policy: "auditor-audit-log",
    };
  }

  // ── Default deny ──────────────────────────────────────────────────────────
  return {
    allowed: false,
    reason: "No matching policy; default deny.",
    policy: POLICY_DENY_DEFAULT,
  };
}

// packages/policies/src/workflow/submitProposal.ts

/**
 * submitProposal service.
 *
 * Creates a Submission, ProposalVersion, ApplicantIdentity, and BlindedPacket
 * in a single Prisma transaction, then writes an AuditEvent.
 *
 * Identity path (ApplicantIdentity) is strictly separated from proposal content
 * (ProposalVersion / BlindedPacket).
 *
 * State transitions produced:
 *   (none → draft)  →  submitted
 *   plus: blinded_packet_created audit event
 */

import { prisma } from "../../domain/src/prisma/client.js";
import { mapToBlindedContent } from "../intake/blindedPacketMapper.js";
import type {
  IntakeRequest,
  SubmitProposalResult,
} from "../../domain/src/entities/index.js";
import type { ProposalVersion as PrismaProposalVersion } from "@prisma/client";

export async function submitProposal(
  request: IntakeRequest,
  actorId: string
): Promise<SubmitProposalResult> {
  const result = await prisma.$transaction(async (tx) => {
    // 1. Derive a pseudonymous applicant alias from the actor ID.
    //    The alias is what appears on Submission; the real identity goes to
    //    ApplicantIdentity (separate table, separate access path).
    const applicantAlias = `applicant:${actorId}`;

    // 2. Create Submission in `submitted` status (P1 flow — no draft step via API)
    const submission = await tx.submission.create({
      data: {
        call_id: request.call_id,
        applicant_alias: applicantAlias,
        status: "submitted",
        submitted_at: new Date(),
      },
    });

    // 3. Create ProposalVersion (version 1, effective)
    const proposalVersion = await tx.proposalVersion.create({
      data: {
        submission_id: submission.id,
        version_number: 1,
        is_effective: true,
        title: request.proposal.title,
        abstract: request.proposal.abstract,
        requested_budget_k_eur: request.proposal.requested_budget_k_eur,
        budget_usage: request.proposal.budget_usage,
        tasks_breakdown: request.proposal.tasks_breakdown,
        attachments_meta: request.proposal.attachments_meta,
      },
    });

    // 4. Create ApplicantIdentity — SEPARATE from proposal content
    await tx.applicantIdentity.create({
      data: {
        submission_id: submission.id,
        legal_name: request.identity.legal_name,
        email: request.identity.email,
        country: request.identity.country,
        organisation: request.identity.organisation,
      },
    });

    // 5. Generate BlindedPacket from proposal version (identity-free)
    const blindedContent = mapToBlindedContent(
      proposalVersion as PrismaProposalVersion
    );

    const blindedPacket = await tx.blindedPacket.create({
      data: {
        proposal_version_id: proposalVersion.id,
        content: blindedContent,
      },
    });

    // 6. Audit: submission_created
    await tx.auditEvent.create({
      data: {
        event_type: "submission_created",
        actor_id: actorId,
        target_type: "Submission",
        target_id: submission.id,
        payload: {
          call_id: request.call_id,
          status: "submitted",
          version_number: 1,
        },
      },
    });

    // 7. Audit: state_transition (draft → submitted, collapsed in P1)
    await tx.auditEvent.create({
      data: {
        event_type: "state_transition",
        actor_id: actorId,
        target_type: "Submission",
        target_id: submission.id,
        payload: {
          from: "draft",
          to: "submitted",
          trigger: "submit",
        },
      },
    });

    // 8. Audit: blinded_packet_created
    await tx.auditEvent.create({
      data: {
        event_type: "blinded_packet_created",
        actor_id: actorId,
        target_type: "BlindedPacket",
        target_id: blindedPacket.id,
        payload: {
          submission_id: submission.id,
          proposal_version_id: proposalVersion.id,
        },
      },
    });

    return {
      submission_id: submission.id,
      proposal_version_id: proposalVersion.id,
      blinded_packet_id: blindedPacket.id,
      status: "submitted" as const,
    };
  });

  return result;
}

/**
 * runEligibilityCheck service.
 *
 * Orchestrates:
 *   1. Fetch submission + call configuration
 *   2. Build EligibilityRuleSet from Call config
 *   3. Run pure evaluateEligibility()
 *   4. Persist EligibilityRecord (in transaction)
 *   5. Transition submission status
 *   6. Write AuditEvent
 *
 * State transitions produced:
 *   submitted → eligible | eligibility_failed
 */

import { prisma } from "../../domain/src/prisma/client.js";
import { evaluateEligibility } from "../eligibility/evaluateEligibility.js";
import type {
  EligibilityInputs,
  RunEligibilityCheckResult,
  EligibilityRuleSet,
} from "../../domain/src/entities/index.js";

// Default rule set version tag — matches mvp-profile.md
const DEFAULT_RULE_SET_VERSION = "v0.6.0";

export async function runEligibilityCheck(
  submissionId: string,
  inputs: EligibilityInputs,
  actorId: string
): Promise<RunEligibilityCheckResult> {
  // 1. Fetch submission with call configuration
  const submission = await prisma.submission.findUnique({
    where: { id: submissionId },
    include: {
      call: {
        select: {
          id: true,
          enabled_eligibility_checks: true,
          max_budget_k_eur: true,
          status: true,
        },
      },
    },
  });

  if (!submission) {
    throw new Error(`Submission not found: ${submissionId}`);
  }

  if (submission.status !== "submitted") {
    throw new Error(
      `Eligibility check can only run on submitted proposals. Current status: ${submission.status}`
    );
  }

  // 2. Build EligibilityRuleSet from Call config
  const ruleSet: EligibilityRuleSet = {
    version: DEFAULT_RULE_SET_VERSION,
    enabled_checks: submission.call.enabled_eligibility_checks,
    max_budget_k_eur: submission.call.max_budget_k_eur,
  };

  // 3. Run pure evaluation
  const evaluationResult = evaluateEligibility(inputs, ruleSet);

  // 4. Persist + transition in transaction
  const result = await prisma.$transaction(async (tx) => {
    // Create EligibilityRecord
    const eligibilityRecord = await tx.eligibilityRecord.create({
      data: {
        submission_id: submissionId,
        status: evaluationResult.status,
        inputs: evaluationResult.inputs as object,
        active_rules: evaluationResult.active_rules,
        failure_reasons: evaluationResult.failure_reasons,
        evaluated_at: new Date(),
      },
    });

    // Determine new submission status
    const newStatus =
      evaluationResult.status === "eligible"
        ? ("eligible" as const)
        : ("eligibility_failed" as const);

    const previousStatus = submission.status;

    // Transition submission status
    await tx.submission.update({
      where: { id: submissionId },
      data: { status: newStatus },
    });

    // Audit: eligibility_evaluated
    await tx.auditEvent.create({
      data: {
        event_type: "eligibility_evaluated",
        actor_id: actorId,
        target_type: "EligibilityRecord",
        target_id: eligibilityRecord.id,
        payload: {
          submission_id: submissionId,
          result: evaluationResult.status,
          failure_reasons: evaluationResult.failure_reasons,
          active_rules: evaluationResult.active_rules,
          rule_set_version: evaluationResult.rule_set_version,
          inputs: evaluationResult.inputs,
        },
      },
    });

    // Audit: state_transition
    await tx.auditEvent.create({
      data: {
        event_type: "state_transition",
        actor_id: actorId,
        target_type: "Submission",
        target_id: submissionId,
        payload: {
          from: previousStatus,
          to: newStatus,
          trigger: "eligibility_check",
          eligibility_record_id: eligibilityRecord.id,
        },
      },
    });

    return {
      eligibility_record_id: eligibilityRecord.id,
      submission_id: submissionId,
      status: evaluationResult.status,
      failure_reasons: evaluationResult.failure_reasons,
    };
  });

  return result;
}

// app/web/src/server/auth/middleware.ts

/**
 * Authentication middleware.
 *
 * For P0-P2 (dev mode), role is resolved from:
 *   1. JWT Bearer token (if present) — verified with JWT_SECRET env var
 *   2. X-Role + X-Actor-Id dev headers (when NODE_ENV !== "production")
 *
 * In production, only JWT is accepted.
 *
 * Returns a Principal or null (unauthenticated).
 */

import { type NextRequest } from "next/server";
import { z } from "zod";
import {
  PrincipalSchema,
  PrincipalRoleSchema,
} from "../../../../../packages/domain/src/schemas/index.js";
import type { Principal } from "../../../../../packages/domain/src/entities/index.js";

// Minimal JWT decode (no external library required for P0-P2 dev mode).
// In production this MUST be replaced with proper verification (e.g. jose).
function decodeJwtPayload(token: string): Record<string, unknown> | null {
  try {
    const parts = token.split(".");
    if (parts.length !== 3) return null;
    const payload = parts[1];
    if (!payload) return null;
    // Base64url → base64
    const base64 = payload.replace(/-/g, "+").replace(/_/g, "/");
    const json = Buffer.from(base64, "base64").toString("utf-8");
    return JSON.parse(json) as Record<string, unknown>;
  } catch {
    return null;
  }
}

const DevHeadersSchema = z.object({
  role: PrincipalRoleSchema,
  actorId: z.string().min(1),
  identity: z.string().optional(),
});

export function extractPrincipal(req: NextRequest): Principal | null {
  // 1. Try JWT Bearer token
  const authHeader = req.headers.get("authorization");
  if (authHeader?.startsWith("Bearer ")) {
    const token = authHeader.slice(7);
    const payload = decodeJwtPayload(token);
    if (payload) {
      const parsed = PrincipalSchema.safeParse({
        id: payload["sub"],
        role: payload["role"],
        identity: payload["identity"],
      });
      if (parsed.success) {
        return parsed.data;
      }
    }
  }

  // 2. Dev-mode header fallback (non-production only)
  if (process.env.NODE_ENV !== "production") {
    const roleHeader = req.headers.get("x-role");
    const actorHeader = req.headers.get("x-actor-id");
    const identityHeader = req.headers.get("x-identity");

    if (roleHeader && actorHeader) {
      const parsed = DevHeadersSchema.safeParse({
        role: roleHeader,
        actorId: actorHeader,
        identity: identityHeader ?? undefined,
      });

      if (parsed.success) {
        return {
          id: parsed.data.actorId,
          role: parsed.data.role,
          identity: parsed.data.identity,
        };
      }
    }
  }

  return null;
}

/**
 * Convenience: extract principal and return 401 response if unauthenticated.
 * Usage: `const { principal, error } = requirePrincipal(req); if (error) return error;`
 */
export function requirePrincipal(req: NextRequest): {
  principal: Principal;
  error: null;
} | {
  principal: null;
  error: Response;
} {
  const principal = extractPrincipal(req);
  if (!principal) {
    return {
      principal: null,
      error: new Response(
        JSON.stringify({ error: "Unauthenticated" }),
        { status: 401, headers: { "Content-Type": "application/json" } }
      ),
    };
  }
  return { principal, error: null };
}

/**
 * Convenience: require a specific role, return 403 if wrong role.
 */
export function requireRole(
  req: NextRequest,
  ...allowedRoles: Principal["role"][]
): { principal: Principal; error: null } | { principal: null; error: Response } {
  const result = requirePrincipal(req);
  if (result.error) return result;

  if (!allowedRoles.includes(result.principal.role)) {
    return {
      principal: null,
      error: new Response(
        JSON.stringify({
          error: "Forbidden",
          detail: `Role '${result.principal.role}' is not permitted for this operation.`,
        }),
        { status: 403, headers: { "Content-Type": "application/json" } }
      ),
    };
  }

  return result;
}

// app/web/src/app/api/intake/route.ts

/**
 * POST /api/intake
 *
 * Creates a new submission (proposal + identity + blinded packet).
 *
 * Authorised principals: applicant, admin
 *
 * Request body: IntakeRequestSchema
 * Response: SubmitProposalResult
 */

import { type NextRequest, NextResponse } from "next/server";
import { IntakeRequestSchema } from "../../../../../../packages/domain/src/schemas/index.js";
import { submitProposal } from "../../../../../../packages/policies/src/workflow/submitProposal.js";
import { checkAccess } from "../../../../../../packages/policies/src/access/checkAccess.js";
import { requirePrincipal } from "../../../server/auth/middleware.js";
import { prisma } from "../../../../../../packages/domain/src/prisma/client.js";

export async function POST(req: NextRequest): Promise<NextResponse> {
  // 1. Authenticate
  const auth = requirePrincipal(req);
  if (auth.error) return NextResponse.json({ error: "Unauthenticated" }, { status: 401 });
  const { principal } = auth;

  // 2. Authorise — only applicants and admins may submit proposals
  const decision = checkAccess({
    principal,
    resource_type: "Submission",
    resource_id: "new",
    operation: "write",
    resource_meta: {
      // For new submissions the applicant_alias equals principal.identity
      applicant_alias: principal.identity ?? principal.id,
    },
  });

  // Log access decision as AuditEvent (fire-and-forget, non-blocking)
  void prisma.auditEvent.create({
    data: {
      event_type: decision.allowed ? "access" : "access_denied",
      actor_id: principal.id,
      target_type: "Submission",
      target_id: "new",
      payload: {
        operation: "write",
        policy: decision.policy,
        reason: decision.reason,
        allowed: decision.allowed,
      },
    },
  }).catch(() => {
    // Audit write failure must not break the user-facing response
  });

  if (!decision.allowed) {
    return NextResponse.json(
      { error: "Forbidden", detail: decision.reason },
      { status: 403 }
    );
  }

  // 3. Validate request body
  let body: unknown;
  try {
    body = await req.json();
  } catch {
    return NextResponse.json({ error: "Invalid JSON body" }, { status: 400 });
  }

  const parsed = IntakeRequestSchema.safeParse(body);
  if (!parsed.success) {
    return NextResponse.json(
      { error: "Validation failed", details: parsed.error.flatten() },
      { status: 422 }
    );
  }

  // 4. Execute service
  try {
    const result = await submitProposal(parsed.data, principal.id);
    return NextResponse.json(result, { status: 201 });
  } catch (err) {
    const message = err instanceof Error ? err.message : "Internal server error";
    return NextResponse.json({ error: message }, { status: 500 });
  }
}

## Gaps

- **[GAP-001]** `specs/access-policies.md` does not specify whether applicants are permitted to call `POST /api/eligibility` or only admins. The spec table in `workflow-states.md` shows admins have `read/write` on `submitted` proposals while applicants have only `read`. Assumed: eligibility check is admin-triggered only. Applicant callers receive `403`.

- **[GAP-002]** The spec does not define the exact JWT payload structure (claims, signing algorithm). The `extractPrincipal` function performs base64 decode only (no signature verification) with a comment noting it must be replaced with `jose` for production. The dev-mode `X-Role` / `X-Actor-Id` header path remains available for testing as required.

- **[GAP-003]** `ApplicantIdentity.updated_at` is present in the domain model but Prisma requires an `@updatedAt` directive, meaning it is set automatically. This is consistent with the spec but the spec does not explicitly call out automatic update semantics — assumed safe.

- **[GAP-004]** `BlindedPacket` is specified to be generated on transition to `eligible` in `workflow-states.md`, but also on initial submission in P1. In this implementation, `BlindedPacket` is created during `submitProposal` (on first submission), which satisfies the P1 intake requirement. The `eligible` transition in P2 does not re-generate the packet since the effective `ProposalVersion` already has one. If a new version is required on eligibility, that would be a P3+ concern.

- **[GAP-005]** The `prisma/client.ts` singleton is placed at `packages/domain/src/prisma/client.ts` for import reuse across packages. Depending on monorepo configuration (`tsconfig paths`, `package.json exports`), import paths may need adjustment to match the actual workspace resolver.
