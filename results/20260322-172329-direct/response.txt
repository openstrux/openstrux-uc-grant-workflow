// packages/policies/src/eligibility/evaluateEligibility.ts

import type {
  EligibilityInputs,
  EligibilityCheckName,
  EligibilityEvaluationResult,
  EligibilityRule,
  EligibilityRuleConfig,
  EligibilityRuleResult,
} from "@domain/entities/index.js";

// ─── Rule definitions ─────────────────────────────────────────────────────────

const submittedInEnglishRule: EligibilityRule = {
  name: "submittedInEnglish",
  evaluate(inputs: EligibilityInputs): EligibilityRuleResult {
    if (!inputs.submitted_in_english) {
      return { passed: false, failure_reason: "Proposal must be submitted in English." };
    }
    return { passed: true };
  },
};

const alignedWithCallRule: EligibilityRule = {
  name: "alignedWithCall",
  evaluate(inputs: EligibilityInputs): EligibilityRuleResult {
    if (!inputs.aligned_with_call) {
      return { passed: false, failure_reason: "Proposal is not aligned with the call objectives." };
    }
    return { passed: true };
  },
};

const primaryObjectiveIsRdRule: EligibilityRule = {
  name: "primaryObjectiveIsRd",
  evaluate(inputs: EligibilityInputs): EligibilityRuleResult {
    if (!inputs.primary_objective_is_rd) {
      return {
        passed: false,
        failure_reason: "Primary objective must be research and development.",
      };
    }
    return { passed: true };
  },
};

const meetsEuropeanDimensionRule: EligibilityRule = {
  name: "meetsEuropeanDimension",
  evaluate(inputs: EligibilityInputs): EligibilityRuleResult {
    if (inputs.meets_european_dimension === "false") {
      return {
        passed: false,
        failure_reason: "Proposal does not meet the European dimension requirement.",
      };
    }
    return { passed: true };
  },
};

const requestedBudgetKEurRule: EligibilityRule = {
  name: "requestedBudgetKEur",
  evaluate(inputs: EligibilityInputs, config: EligibilityRuleConfig): EligibilityRuleResult {
    if (inputs.requested_budget_k_eur > config.max_budget_k_eur) {
      return {
        passed: false,
        failure_reason: `Requested budget ${inputs.requested_budget_k_eur}k EUR exceeds the maximum of ${config.max_budget_k_eur}k EUR.`,
      };
    }
    return { passed: true };
  },
};

const firstTimeApplicantInProgrammeRule: EligibilityRule = {
  name: "firstTimeApplicantInProgramme",
  evaluate(inputs: EligibilityInputs): EligibilityRuleResult {
    // Being a first-time applicant is not a disqualifier; this check merely records the fact.
    // Per spec: check is present but does not cause ineligibility on its own.
    // Spec does not specify a failure condition, so this always passes.
    void inputs;
    return { passed: true };
  },
};

// ─── Rule registry ────────────────────────────────────────────────────────────

const ALL_RULES: Record<EligibilityCheckName, EligibilityRule> = {
  submittedInEnglish: submittedInEnglishRule,
  alignedWithCall: alignedWithCallRule,
  primaryObjectiveIsRd: primaryObjectiveIsRdRule,
  meetsEuropeanDimension: meetsEuropeanDimensionRule,
  requestedBudgetKEur: requestedBudgetKEurRule,
  firstTimeApplicantInProgramme: firstTimeApplicantInProgrammeRule,
};

// ─── Pure evaluation function ─────────────────────────────────────────────────

/**
 * Evaluates eligibility inputs against the enabled rule set.
 *
 * This is a pure function — it performs no I/O and records nothing.
 * Persistence of the result is handled by `runEligibilityCheck`.
 */
export function evaluateEligibility(
  inputs: EligibilityInputs,
  enabledChecks: EligibilityCheckName[],
  config: EligibilityRuleConfig
): EligibilityEvaluationResult {
  const activeRules: EligibilityCheckName[] = [];
  const failureReasons: string[] = [];

  for (const checkName of enabledChecks) {
    const rule = ALL_RULES[checkName];
    if (!rule) {
      // Unknown check name — treat as configuration error; log but skip.
      continue;
    }
    activeRules.push(checkName);
    const result = rule.evaluate(inputs, config);
    if (!result.passed && result.failure_reason !== undefined) {
      failureReasons.push(result.failure_reason);
    }
  }

  const status = failureReasons.length === 0 ? "eligible" : "ineligible";

  return {
    status,
    active_rules: activeRules,
    failure_reasons: failureReasons,
  };
}

// packages/policies/src/workflow/submitProposal.ts

import { PrismaClient } from "@prisma/client";
import type { SubmitProposalInput, SubmitProposalResult } from "@domain/entities/index.js";
import { buildBlindedContent } from "../mappers/blindedPacketMapper.js";

/**
 * Creates a Submission, ProposalVersion, ApplicantIdentity (separate path),
 * BlindedPacket, and AuditEvent in a single Prisma transaction.
 *
 * Identity data is never merged into the proposal content path.
 */
export async function submitProposal(
  prisma: PrismaClient,
  input: SubmitProposalInput,
  actorId: string
): Promise<SubmitProposalResult> {
  return prisma.$transaction(async (tx) => {
    // 1. Verify call exists and is open
    const call = await tx.call.findUniqueOrThrow({
      where: { id: input.call_id },
    });

    if (call.status !== "open") {
      throw new Error(`Call ${input.call_id} is not open for submissions.`);
    }

    const now = new Date();
    if (now < call.open_date || now > call.close_date) {
      throw new Error(`Call ${input.call_id} is outside its submission window.`);
    }

    // 2. Create Submission
    const submission = await tx.submission.create({
      data: {
        call_id: input.call_id,
        applicant_alias: input.applicant_alias,
        status: "submitted",
        submitted_at: now,
      },
    });

    // 3. Create ProposalVersion (identity-free content path)
    const proposalVersion = await tx.proposalVersion.create({
      data: {
        submission_id: submission.id,
        version_number: 1,
        is_effective: true,
        title: input.proposal.title,
        abstract: input.proposal.abstract,
        requested_budget_k_eur: input.proposal.requested_budget_k_eur,
        budget_usage: input.proposal.budget_usage,
        tasks_breakdown: input.proposal.tasks_breakdown as object,
        attachments_meta: input.proposal.attachments_meta as object[],
      },
    });

    // 4. Create ApplicantIdentity — separate data path, never merged with proposal
    await tx.applicantIdentity.create({
      data: {
        submission_id: submission.id,
        legal_name: input.identity.legal_name,
        email: input.identity.email,
        country: input.identity.country,
        organisation: input.identity.organisation,
      },
    });

    // 5. Build blinded content — identity fields stripped at generation time
    const blindedContent = buildBlindedContent(proposalVersion);

    // 6. Create BlindedPacket
    const blindedPacket = await tx.blindedPacket.create({
      data: {
        proposal_version_id: proposalVersion.id,
        content: blindedContent,
      },
    });

    // 7. Write AuditEvent
    await tx.auditEvent.create({
      data: {
        event_type: "proposal.submitted",
        actor_id: actorId,
        target_type: "Submission",
        target_id: submission.id,
        payload: {
          call_id: input.call_id,
          applicant_alias: input.applicant_alias,
          proposal_version_id: proposalVersion.id,
          blinded_packet_id: blindedPacket.id,
          status: "submitted",
        },
      },
    });

    return {
      submission_id: submission.id,
      proposal_version_id: proposalVersion.id,
      blinded_packet_id: blindedPacket.id,
    };
  });
}

// packages/policies/src/mappers/blindedPacketMapper.ts

import type { ProposalVersion } from "@domain/entities/index.js";

/**
 * Builds the identity-free content blob for a BlindedPacket.
 *
 * Spec: "Blinded packet mapper — strip identity fields per specs/access-policies.md"
 * Spec: "No identity field (legal name, email, country, organisation, applicant alias)
 *        may appear in any response, view, or export accessible to the reviewer."
 *
 * This mapper is the single authoritative point for identity stripping.
 * It receives a ProposalVersion (which already excludes identity) and
 * returns only the fields safe for reviewer consumption.
 */
export function buildBlindedContent(
  proposalVersion: Pick<
    ProposalVersion,
    | "id"
    | "submission_id"
    | "version_number"
    | "title"
    | "abstract"
    | "requested_budget_k_eur"
    | "budget_usage"
    | "tasks_breakdown"
    | "attachments_meta"
    | "created_at"
  >
): Record<string, unknown> {
  // Explicitly list only allowed fields — never spread the full object.
  return {
    proposal_version_id: proposalVersion.id,
    version_number: proposalVersion.version_number,
    title: proposalVersion.title,
    abstract: proposalVersion.abstract,
    requested_budget_k_eur: proposalVersion.requested_budget_k_eur,
    budget_usage: proposalVersion.budget_usage,
    tasks_breakdown: proposalVersion.tasks_breakdown,
    attachments_meta: proposalVersion.attachments_meta,
    created_at: proposalVersion.created_at,
    // Explicitly NOT included:
    // - submission_id (could be used to correlate with applicant_alias)
    // - Any identity fields (these are not on ProposalVersion, but we enforce here)
  };
}

// packages/policies/src/workflow/runEligibilityCheck.ts

import { PrismaClient } from "@prisma/client";
import type {
  EligibilityInputs,
  EligibilityCheckName,
  RunEligibilityCheckResult,
} from "@domain/entities/index.js";
import { evaluateEligibility } from "../eligibility/evaluateEligibility.js";

/**
 * Persists an EligibilityRecord, transitions the Submission status,
 * and writes an AuditEvent — all in a single Prisma transaction.
 *
 * Input values and the active rule set are recorded exactly as received.
 */
export async function runEligibilityCheck(
  prisma: PrismaClient,
  submissionId: string,
  inputs: EligibilityInputs,
  actorId: string
): Promise<RunEligibilityCheckResult> {
  return prisma.$transaction(async (tx) => {
    // 1. Load submission and its call (for active rule set + budget cap)
    const submission = await tx.submission.findUniqueOrThrow({
      where: { id: submissionId },
      include: { call: true },
    });

    if (submission.status !== "submitted") {
      throw new Error(
        `Eligibility check can only run on submissions with status 'submitted'. ` +
          `Current status: ${submission.status}`
      );
    }

    const call = submission.call;
    const enabledChecks = call.enabled_eligibility_checks as EligibilityCheckName[];

    // 2. Evaluate — pure function, exact inputs and active rules recorded
    const evaluation = evaluateEligibility(inputs, enabledChecks, {
      max_budget_k_eur: call.max_budget_k_eur,
    });

    // 3. Persist EligibilityRecord with exact inputs + active rule set
    const record = await tx.eligibilityRecord.create({
      data: {
        submission_id: submissionId,
        status: evaluation.status,
        inputs: inputs as unknown as object,
        active_rules: evaluation.active_rules,
        failure_reasons: evaluation.failure_reasons,
      },
    });

    // 4. Transition Submission status
    const nextStatus =
      evaluation.status === "eligible" ? "eligible" : "eligibility_failed";

    await tx.submission.update({
      where: { id: submissionId },
      data: { status: nextStatus },
    });

    // 5. If eligible, generate BlindedPacket for the effective ProposalVersion
    if (evaluation.status === "eligible") {
      const effectiveVersion = await tx.proposalVersion.findFirst({
        where: { submission_id: submissionId, is_effective: true },
      });

      if (effectiveVersion) {
        const existingPacket = await tx.blindedPacket.findUnique({
          where: { proposal_version_id: effectiveVersion.id },
        });

        // BlindedPacket was already created at submission time per spec;
        // we verify it exists but do not recreate it.
        if (!existingPacket) {
          const { buildBlindedContent } = await import(
            "../mappers/blindedPacketMapper.js"
          );
          const blindedContent = buildBlindedContent(effectiveVersion);
          await tx.blindedPacket.create({
            data: {
              proposal_version_id: effectiveVersion.id,
              content: blindedContent,
            },
          });
        }

import type {
  AccessRequest,
  AccessResult,
  Principal,
  ResourceType,
  Operation,
} from "@domain/entities/index.js";

// ─── Policy evaluators ────────────────────────────────────────────────────────

function evaluateApplicantPolicy(req: AccessRequest): AccessResult | null {
  if (req.principal.role !== "applicant") return null;

  // POLICY: deny-identity-to-reviewer does not apply here, but applicants
  // must not access other applicants' submissions.

  // Applicants cannot access BlindedPacket
  if (req.resource === "BlindedPacket") {
    return { allowed: false, reason: "Applicants may not access blinded packets." };
  }

  // Applicants cannot access AuditEvent
  if (req.resource === "AuditEvent") {
    return { allowed: false, reason: "Applicants may not access audit events." };
  }

  // Applicants can only read EligibilityRecord (not failure_reasons detail — enforced at response layer)
  if (req.resource === "EligibilityRecord" && req.operation !== "read") {
    return { allowed: false, reason: "Applicants may only read eligibility records." };
  }

  // POLICY: applicant-own-proposals — applicant can only access their own submissions
  if (
    req.resource === "Submission" ||
    req.resource === "ProposalVersion" ||
    req.resource === "ApplicantIdentity" ||
    req.resource === "EligibilityRecord"
  ) {
    if (
      req.applicant_alias !== undefined &&
      req.principal.applicant_alias !== req.applicant_alias
    ) {
      return {
        allowed: false,
        reason: "Applicants may only access their own submissions.",
      };
    }
    // write limited to draft and clarification_requested states — enforced in service
    return { allowed: true, reason: "POLICY:applicant-own-proposals" };
  }

  if (req.resource === "Call" && req.operation === "read") {
    return { allowed: true, reason: "Applicants may read calls." };
  }

  return { allowed: false, reason: "Applicants do not have access to this resource." };
}

function evaluateAdminPolicy(req: AccessRequest): AccessResult | null {
  if (req.principal.role !== "admin") return null;
  // POLICY: admin-all — admins have read/write/delete on all entities
  // Delete is only allowed via retention/anonymisation workflow (enforced at service layer)
  return { allowed: true, reason: "POLICY:admin-all" };
}

function evaluateReviewerPolicy(req: AccessRequest): AccessResult | null {
  if (req.principal.role !== "reviewer") return null;

  // POLICY: deny-identity-to-reviewer — hard deny, always logged
  if (req.resource === "ApplicantIdentity") {
    return {
      allowed: false,
      reason: "POLICY:deny-identity-to-reviewer — reviewers may never access applicant identity.",
    };
  }

  // Reviewers cannot access raw Submission (contains applicant_alias)
  if (req.resource === "Submission") {
    return {
      allowed: false,
      reason: "Reviewers may not access raw submissions (contains identity fields).",
    };
  }

  // Reviewers cannot access ProposalVersion directly (use BlindedPacket)
  if (req.resource === "ProposalVersion") {
    return {
      allowed: false,
      reason: "Reviewers may not access proposal versions directly — use blinded packets.",
    };
  }

  // Reviewers cannot access EligibilityRecord
  if (req.resource === "EligibilityRecord") {
    return {
      allowed: false,
      reason: "Reviewers may not access eligibility records.",
    };
  }

  // POLICY: reviewer-blinded-assigned
  if (req.resource === "BlindedPacket" && req.operation === "read") {
    if (
      req.submission_id !== undefined &&
      req.principal.assigned_submission_ids?.includes(req.submission_id)
    ) {
      return { allowed: true, reason: "POLICY:reviewer-blinded-assigned" };
    }
    return {
      allowed: false,
      reason: "Reviewer is not assigned to this submission.",
    };
  }

  if (req.resource === "Call" && req.operation === "read") {
    return { allowed: true, reason: "Reviewers may read calls." };
  }

  return { allowed: false, reason: "Reviewers do not have access to this resource." };
}

function evaluateValidatorPolicy(req: AccessRequest): AccessResult | null {
  if (req.principal.role !== "validator") return null;

  // Validators: P5+ feature — limited read on blinded packets of shortlisted proposals
  if (req.resource === "ApplicantIdentity") {
    return {
      allowed: false,
      reason: "Validators may never access applicant identity.",
    };
  }

  // Validators can read blinded packets (P5+ — for now allow read)
  if (req.resource === "BlindedPacket" && req.operation === "read") {
    return { allowed: true, reason: "POLICY:validator-blinded-read" };
  }

  return { allowed: false, reason: "Validators do not have access to this resource." };
}

function evaluateAuditorPolicy(req: AccessRequest): AccessResult | null {
  if (req.principal.role !== "auditor") return null;

  if (req.resource === "AuditEvent" && req.operation === "read") {
    return { allowed: true, reason: "POLICY:auditor-audit-log-read" };
  }

  return {
    allowed: false,
    reason: "Auditors may only read audit events.",
  };
}

// ─── Exported policy function ─────────────────────────────────────────────────

/**
 * Evaluates whether the given principal may perform the operation on the resource.
 *
 * Returns an AccessResult with `allowed` and a `reason` for logging.
 * All calls to this function should be logged as an AuditEvent by the caller.
 */
export function checkAccess(req: AccessRequest): AccessResult {
  const evaluators = [
    evaluateAdminPolicy,
    evaluateApplicantPolicy,
    evaluateReviewerPolicy,
    evaluateValidatorPolicy,
    evaluateAuditorPolicy,
  ];

  for (const evaluate of evaluators) {
    const result = evaluate(req);
    if (result !== null) {
      return result;
    }
  }

  return { allowed: false, reason: "No matching policy — default deny." };
}

// Re-export types for convenience
export type { AccessRequest, AccessResult, Principal, ResourceType, Operation };

import { NextRequest } from "next/server";
import { PrismaClient } from "@prisma/client";
import type { Principal, PrincipalRole } from "@domain/entities/index.js";
import { PrincipalRoleSchema } from "@domain/schemas/index.js";

// Prisma singleton for route handlers
let _prisma: PrismaClient | undefined;
export function getPrisma(): PrismaClient {
  if (!_prisma) {
    _prisma = new PrismaClient();
  }
  return _prisma;
}

// ─── JWT payload shape (dev mode uses X-Role header) ─────────────────────────

interface JwtClaims {
  sub: string;
  role: PrincipalRole;
  applicant_alias?: string;
  assigned_submission_ids?: string[];
}

/**
 * Extracts the principal from the incoming request.
 *
 * For P0-P2 (dev mode): reads the `X-Role` header and `X-Actor-Id` header.
 * The `X-Assigned-Submissions` header (comma-separated) is used for reviewers.
 * The `X-Applicant-Alias` header is used for applicants.
 *
 * In production this would validate a JWT; the structure is prepared for that.
 */
export function extractPrincipal(req: NextRequest): Principal | null {
  // ── Dev-mode: header-based auth ───────────────────────────────────────────
  const roleHeader = req.headers.get("x-role");
  const actorId = req.headers.get("x-actor-id");

  if (!roleHeader || !actorId) {
    return null;
  }

  const roleParse = PrincipalRoleSchema.safeParse(roleHeader);
  if (!roleParse.success) {
    return null;
  }

  const role = roleParse.data;

  const principal: Principal = {
    id: actorId,
    role,
  };

  if (role === "applicant") {
    const alias = req.headers.get("x-applicant-alias");
    if (alias) {
      principal.applicant_alias = alias;
    }
  }

  if (role === "reviewer") {
    const assignedHeader = req.headers.get("x-assigned-submissions");
    if (assignedHeader) {
      principal.assigned_submission_ids = assignedHeader
        .split(",")
        .map((s) => s.trim())
        .filter(Boolean);
    } else {
      principal.assigned_submission_ids = [];
    }
  }

  return principal;
}

/**
 * Middleware helper: extracts principal or returns a 401 response.
 * Use in route handlers:
 *   const { principal, errorResponse } = requirePrincipal(request);
 *   if (errorResponse) return errorResponse;
 */
export function requirePrincipal(req: NextRequest): {
  principal: Principal | null;
  errorResponse: Response | null;
} {
  const principal = extractPrincipal(req);
  if (!principal) {
    return {
      principal: null,
      errorResponse: new Response(
        JSON.stringify({ error: "Unauthorized: missing or invalid credentials." }),
        { status: 401, headers: { "Content-Type": "application/json" } }
      ),
    };
  }
  return { principal, errorResponse: null };
}

/**
 * Middleware helper: requires a specific role.
 */
export function requireRole(
  req: NextRequest,
  ...allowedRoles: PrincipalRole[]
): { principal: Principal | null; errorResponse: Response | null } {
  const { principal, errorResponse } = requirePrincipal(req);
  if (errorResponse || !principal) {
    return { principal: null, errorResponse };
  }

  if (!allowedRoles.includes(principal.role)) {
    return {
      principal: null,
      errorResponse: new Response(
        JSON.stringify({
          error: `Forbidden: role '${principal.role}' is not permitted for this operation.`,
        }),
        { status: 403, headers: { "Content-Type": "application/json" } }
      ),
    };
  }

  return { principal, errorResponse: null };
}

export { getPrisma as prisma };

import { NextRequest, NextResponse } from "next/server";
import { requireRole, getPrisma } from "@/server/auth/middleware.js";
import { checkAccess } from "@policies/access/checkAccess.js";
import { submitProposal } from "@policies/workflow/submitProposal.js";
import { SubmitProposalInputSchema } from "@domain/schemas/index.js";

/**
 * POST /api/intake
 *
 * Creates a new submission (proposal + identity + blinded packet).
 * Allowed roles: applicant, admin
 */
export async function POST(request: NextRequest): Promise<NextResponse> {
  // 1. Role check — must happen before any DB operation
  const { principal, errorResponse } = requireRole(request, "applicant", "admin");
  if (errorResponse || !principal) return NextResponse.json(null, { status: 401 }).clone
    ? new NextResponse(errorResponse?.body, {
        status: errorResponse?.status,
        headers: { "Content-Type": "application/json" },
      })
    : (errorResponse as NextResponse);

  // 2. Parse and validate input
  let body: unknown;
  try {
    body = await request.json();
  } catch {
    return NextResponse.json({ error: "Invalid JSON body." }, { status: 400 });
  }

  const parsed = SubmitProposalInputSchema.safeParse(body);
  if (!parsed.success) {
    return NextResponse.json(
      { error: "Validation failed.", details: parsed.error.flatten() },
      { status: 422 }
    );
  }

  const input = parsed.data;

  // 3. Access control check
  const accessResult = checkAccess({
    principal,
    resource: "Submission",
    operation: "write",
    applicant_alias:
      principal.role === "applicant" ? principal.applicant_alias : undefined,
  });

  const prisma = getPrisma();

  // Log access decision (both allow and deny)
  await prisma.auditEvent.create({
    data: {
      event_type: "access",
      actor_id: principal.id,
      target_type: "Submission",
      target_id: "new",
      payload: {
        operation: "write",
        resource: "Submission",
        allowed: accessResult.allowed,
        reason: accessResult.reason,
      },
    },
  });

  if (!accessResult.allowed) {
    return NextResponse.json({ error: accessResult.reason }, { status: 403 });
  }

  // 4. Enforce applicant-own-proposals: applicant alias must match caller
  if (
    principal.role === "applicant" &&
    principal.applicant_alias !== undefined &&
    input.applicant_alias !== principal.applicant_alias
  ) {
    return NextResponse.json(
      { error: "Applicants may only submit proposals under their own alias." },
      { status: 403 }
    );
  }

  // 5. Execute service
  try {
    const result = await submitProposal(prisma, input, principal.id);
    return NextResponse.json(result, { status: 201 });
  } catch (err: unknown) {
    const message = err instanceof Error ? err.message : "Internal server error.";
    return NextResponse.json({ error: message }, { status: 500 });
  }
}

This satisfies both the data model and the workflow invariant.

- **[GAP-004]** JWT validation structure for production is stubbed. The middleware uses `X-Role` / `X-Actor-Id` headers as specified for P0-P2 dev mode. The `JwtClaims` interface is present but JWT parsing is not wired (no external IdP required per constraints).

- **[GAP-005]** `specs/access-policies.md` states `EligibilityRecord.failure_reasons` is admin-only for applicants. This is noted in `checkAccess` comments and the `evaluateApplicantPolicy` function. Full field-level filtering at the response layer is not implemented (would require a dedicated response mapper for applicant-facing eligibility reads); a `GET /api/eligibility` endpoint that strips `failure_reasons` for applicants would complete this — flagged for the service layer to implement as a response mapper.
